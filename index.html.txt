  <button onclick="setOperation('Delete')">Delete</button>
  <button onclick="setOperation('Query')">Query</button>

  <p>Operation: <span id="operation">None</span></p>

  <input id="prompt" placeholder="Enter prompt here">

  <div style="margin-top:10px;">
    <button onclick="submitToAI()">Run</button>
  </div>

  <div id="status"></div>

  <textarea id="response" placeholder="Response will appear here"></textarea>

<script>
  let widgetState = {
    operation: null,
    prompt: "",
    response: ""
  };

  function setOperation(op) {
    widgetState.operation = op;
    document.getElementById("operation").innerText = op;
  }

  async function submitToAI() {
    const opSpan = document.getElementById("operation");
    const promptInput = document.getElementById("prompt");
    const responseBox = document.getElementById("response");
    const status = document.getElementById("status");

    const operation = opSpan.innerText;
    const prompt = promptInput.value.trim();

    if (!operation || operation === "None") {
      status.innerText = "Select an operation first.";
      return;
    }

    if (!prompt) {
      status.innerText = "Enter a prompt before running.";
      return;
    }

    status.innerText = "Sending to AI...";
    responseBox.value = "";

    // Update local state
    widgetState.operation = operation;
    widgetState.prompt = prompt;

    try {
      // TODO: replace this URL with your real AI endpoint
      const apiUrl = "https://your-ai-endpoint.example.com/run";

      const body = {
        operation: widgetState.operation,
        prompt: widgetState.prompt
      };

      const res = await fetch(apiUrl, {
        method: "POST",
        headers: {
          "Content-Type": "application/json"
          // Add auth headers here if needed
        },
        body: JSON.stringify(body)
      });

      if (!res.ok) {
        throw new Error("HTTP " + res.status);
      }

      const data = await res.json();

      // Adjust this depending on your API’s response shape
      const aiText = data.result || data.output || JSON.stringify(data, null, 2);

      widgetState.response = aiText;
      responseBox.value = aiText;
      status.innerText = "Done.";
    } catch (err) {
      status.innerText = "Error talking to AI: " + err.message;
      responseBox.value = "";
    }
  }
</script>
